

CREATE TABLE cinemaRooms (
  id bigint NOT NULL PRIMARY KEY,
  headquarterID bigint,
  diagramRoom varchar(500)
);


CREATE TABLE reservations (
  id bigint NOT NULL PRIMARY KEY,
  reservationDate date,
  userID bigint,
  cinemaShowID bigint
);


CREATE TABLE adresses (
  id bigint NOT NULL PRIMARY KEY,
  streetName varchar(500),
  houseNumber bigint,
  city varchar(500),
  state varchar(500),
  country varchar(500),
  postalCode int
);


CREATE TABLE cinemaChains (
  id bigint NOT NULL PRIMARY KEY,
  name varchar(500) NOT NULL
);


CREATE TABLE tokens (
  id bigint NOT NULL PRIMARY KEY,
  accessToken varchar(500) NOT NULL,
  refreshToken varchar(500) NOT NULL,
  createdAt date NOT NULL,
  expiresAt date NOT NULL,
  userID bigint
);


CREATE TABLE reservedSeats (
  id bigint NOT NULL PRIMARY KEY,
  reservationID bigint,
  seatID bigint
);


CREATE TABLE seats (
  id bigint NOT NULL PRIMARY KEY,
  row enum NOT NULL,
  number smallint NOT NULL,
  cinemaRoomID bigint NOT NULL
);


CREATE TABLE cinemaHeadquarters (
  id bigint NOT NULL PRIMARY KEY,
  cinemaChainID bigint,
  headquarterAdressID bigint
);


CREATE TABLE movies (
  id bigint NOT NULL PRIMARY KEY,
  title varchar(500) NOT NULL,
  description varchar(500) NOT NULL,
  poster varchar(500) NOT NULL,
  genre enum NOT NULL
);


CREATE TABLE users (
  id bigint NOT NULL PRIMARY KEY,
  email varchar(500) NOT NULL,
  name varchar(500) NOT NULL,
  password varchar(500)
);


CREATE TABLE cinemaShows (
  id bigint NOT NULL PRIMARY KEY,
  showDate date,
  movieID bigint,
  cinemaRoomID bigint
);
